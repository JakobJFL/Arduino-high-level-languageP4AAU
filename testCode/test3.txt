func loop()
{
    while (shouldBlink())
    {
        ledPin1.Write(HIGH);
        ledPin2.Write(LOW);
        whileWait(2000, shouldBlink);
        ledPin1.Write(LOW);
        ledPin2.Write(HIGH);
        whileWait(2000, shouldBlink);
    }
    ledPin1.Write(LOW);
    ledPin2.Write(LOW);
    Num timesToRun1 = analogPin1.ReadAnalog();
    Num timesToRun2 = analogPin2.ReadAnalog();
    Bool isPressed = shouldRunFunc();
}

func Num shouldBlink() {
    Pwm outputValue = sensorPin.ReadPwm();
    if (outputValue > 128)
    {
        return false;
    }
    return true;
}

func Bool shouldRunFunc(){
    return buttonPin.ReadDigital();
}
